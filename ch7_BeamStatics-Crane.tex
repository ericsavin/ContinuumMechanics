\begin{frame}{Move crane}{Solution}

\begin{exampleblock}{Question \#1: $\Fres_L$? $\Mres_L$?}
\begin{itemize}
\item Equilibrium of the plate:
\begin{displaymath}
\begin{split}
-\Fres_L+m\gv &=\bzero \,,\\
-\Mres_L+(\xv_{C_m}-\xv_P)\times m\gv &=\bzero\,;
\end{split}
\end{displaymath}
\item But $\gv=-g\iv_\zj$ hence:
\begin{displaymath}
\begin{split}
